<!DOCTYPE html>

<html lang="ko">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>카카오 주소 검색</title>

    <script src="https://t1.daumcdn.net/mapjsapi/bundle/postcode/prod/postcode.v2.js"></script>

    <script>

        function openDaumPostcode() {

            var element_wrap = document.getElementById('wrap');


            // ✅ 주소 검색 창을 modal 형식으로 표시

            new daum.Postcode({

                oncomplete: function(data) {

                    // ✅ FlutterFlow WebView로 선택한 주소 전송

                    if (window.ReactNativeWebView) {

                        window.ReactNativeWebView.postMessage(data.roadAddress);

                    } else {

                        alert("WebView 감지 실패: " + data.roadAddress);

                    }


                    // ✅ 모달 닫기

                    element_wrap.style.display = 'none';

                },

                onclose: function(state) {

                    if (state === 'COMPLETE_CLOSE') {

                        element_wrap.style.display = 'none';

                    }

                }

            }).embed(element_wrap);


            // ✅ 모달 표시

            element_wrap.style.display = 'block';

        }

    </script>

    <style>

        #wrap {

            display: none;

            position: fixed;

            top: 50%;

            left: 50%;

            transform: translate(-50%, -50%);

            width: 300px;

            height: 400px;

            background: white;

            z-index: 100;

            border: 1px solid #ccc;

            box-shadow: 0px 0px 10px rgba(0,0,0,0.1);

        }

    </style>

</head>

<body>

    <button onclick="openDaumPostcode()">주소 검색</button>

    <div id="wrap"></div>

</body>

</html>
