<!DOCTYPE html>

<html lang="ko">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>카카오 주소 검색</title>

    <script src="https://t1.daumcdn.net/mapjsapi/bundle/postcode/prod/postcode.v2.js"></script>

    <script>

        function openDaumPostcode(event) {

            if (event) {

                event.preventDefault();  // ✅ FlutterFlow에서 새로고침 방지

            }

            

            new daum.Postcode({

                oncomplete: function(data) {

                    // ✅ 선택한 주소를 FlutterFlow WebView로 전송

                    if (window.ReactNativeWebView) {

                        window.ReactNativeWebView.postMessage(data.roadAddress);

                    } else {

                        alert("WebView 감지 실패: " + data.roadAddress);

                    }

                }

            }).open();

        }

        

        // ✅ WebView에서 기본 이벤트 차단 (FlutterFlow 새로고침 방지)

        document.addEventListener("click", function(event) {

            event.preventDefault();

        });


        // ✅ 페이지 로드 시 자동 실행 제거 (버튼 클릭 시에만 실행)

    </script>

</head>

<body>

    <button onclick="openDaumPostcode(event)">주소 검색</button>

</body>

</html>
